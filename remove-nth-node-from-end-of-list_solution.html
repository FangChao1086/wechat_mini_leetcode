struct ListNode {
    int val;
    ListNode *next;
    ListNode() : val(0), next(nullptr) {}
    ListNode(int x) : val(x), next(nullptr) {}
    ListNode(int x, ListNode *next) : val(x), next(next) {}
};
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* dummy = new ListNode(0, head); // è™šæ‹Ÿå¤´èŠ‚ç‚¹ï¼Œå¤„ç†åˆ é™¤å¤´èŠ‚ç‚¹çš„æƒ…å†µ
        ListNode* fast = dummy;
        ListNode* slow = dummy;
        
        // å¿«æŒ‡é’ˆå…ˆèµ°næ­¥
        for (int i = 0; i < n; i++) {
            fast = fast->next;
        }
        
        // å¿«æ…¢æŒ‡é’ˆåŒæ—¶ç§»åŠ¨ï¼Œç›´åˆ°å¿«æŒ‡é’ˆåˆ°è¾¾æœ«å°¾
        while (fast->next != nullptr) {

            slow = slow->next;
        }
        
        // åˆ é™¤å€’æ•°ç¬¬nä¸ªèŠ‚ç‚¹

        delete dummy;
        return result;
    }
};
