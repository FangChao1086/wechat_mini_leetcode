class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> res;
        vector<int> path;
        dfs(nums, 0, path, res);
        return res;
    }
    void dfs(vector<int>& nums, int idx, vector<int>& path, vector<vector<int>>& res) {
        if (idx == nums.size()) {
            res.push_back(path); // è®°å½•å½“å‰å­é›†
