class Solution {
public:
    int findTargetSumWays(vector<int>& nums, int target) {
        int sum = accumulate(nums.begin(), nums.end(), 0);
        if (target > sum || target < -sum) return 0; // ç›®æ ‡å€¼è¶…å‡ºå¯èƒ½èŒƒå›´
        
        int n = nums.size(), offset = sum; // åç§»é‡å¤„ç†è´Ÿä¸‹æ ‡
        vector<vector<int>> dp(n+1, vector<int>(2*sum+1, 0));
        dp[0][offset] = 1; // åˆå§‹çŠ¶æ€ï¼šä¸é€‰ä»»ä½•æ•°æ—¶å’Œä¸º0
        
        for (int i = 1; i <= n; i++) {
            for (int j = -sum; j <= sum; j++) {
                int idx = j + offset; // å®é™…æ•°ç»„ä¸‹æ ‡
                if (j - nums[i-1] >= -sum) 
                    dp[i][idx] += dp[i-1][idx - nums[i-1]]; // é€‰æ‹©+å·
                if (j + nums[i-1] <= sum) 
                    dp[i][idx] += dp[i-1][idx + nums[i-1]]; // é€‰æ‹©-å·
            }
        }
        return dp[n][target + offset];
    }
};
