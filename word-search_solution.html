class Solution {
public:
    bool exist(vector<vector<char>>& board, string word) {
        int m = board.size(), n = board[0].size();
        vector<vector<bool>> visited(m, vector<bool>(n, false));
        // ä»æ¯ä¸ªæ ¼å­å¼€å§‹å°è¯•æœç´¢
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (dfs(board, word, visited, i, j, 0)) return true;
            }
        }
        return false;
    }
    
    bool dfs(vector<vector<char>>& board, string& word, vector<vector<bool>>& visited, int i, int j, int idx) {
        // å•è¯å…¨éƒ¨åŒ¹é…å®Œæˆ
        if (idx == word.size()) return true;
        // è¶Šç•Œæˆ–å·²è®¿é—®æˆ–å­—ç¬¦ä¸åŒ¹é…
        if (i < 0 || i >= board.size() || j < 0 || j >= board[0].size() 
            || visited[i][j] || board[i][j] != word[idx]) return false;
        
        visited[i][j] = true; // æ ‡è®°è®¿é—®
        // å‘å››ä¸ªæ–¹å‘é€’å½’æœç´¢
        bool found = dfs(board, word, visited, i+1, j, idx+1)
                  || dfs(board, word, visited, i-1, j, idx+1)
                  || dfs(board, word, visited, i, j+1, idx+1)
                  || dfs(board, word, visited, i, j-1, idx+1);
        visited[i][j] = false; // å›æº¯ï¼Œæ’¤é”€æ ‡è®°
        return found;
    }
};
