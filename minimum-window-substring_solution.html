    string minWindow(string s, string t) {
        unordered_map<char, int> need; // ç»Ÿè®¡tä¸­å­—ç¬¦éœ€æ±‚

        int start = 0, len = INT_MAX; // è®°å½•æœ€å°å­ä¸²èµ·å§‹ä½ç½®å’Œé•¿åº¦
        
        unordered_map<char, int> window; // çª—å£å†…å­—ç¬¦è®¡æ•°

            // å½“çª—å£æ»¡è¶³æ‰€æœ‰å­—ç¬¦éœ€æ±‚æ—¶ï¼Œæ”¶ç¼©å·¦è¾¹ç•Œ
            while (valid == need.size()) {
